    <script>
        let currentSearchEngine = 'google';
        
        const searchUrls = {
            'google': 'https://www.google.com/search?q=',
            'images': 'https://www.google.com/search?tbm=isch&q=',
            'video': 'https://www.google.com/search?tbm=vid&q=',
            'news': 'https://www.google.com/search?tbm=nws&q='
        };

        const placeholders = {
            'google': 'Search the web...',
            'images': 'Search for images...',
            'video': 'Search for videos...',
            'news': 'Search latest news...'
        };

        document.getElementById("mainInput").addEventListener("keypress", function(event) {
            if (event.key === "Enter") {
                performSearch();
            }
        });

        function setEngine(engine, element) {
            currentSearchEngine = engine;
            document.querySelectorAll('.tab-btn').forEach(btn => btn.classList.remove('active'));
            element.classList.add('active');
            document.getElementById('mainInput').placeholder = placeholders[engine];
        }

        function performSearch() {
            const query = document.getElementById('mainInput').value;
            if (!query) return;
            const finalUrl = searchUrls[currentSearchEngine] + encodeURIComponent(query);
            window.open(finalUrl, '_blank');
        }

        function quickLink(url) {
            window.open(url, '_blank');
        }
    </script>
